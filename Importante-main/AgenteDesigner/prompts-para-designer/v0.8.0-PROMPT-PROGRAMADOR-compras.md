## PASO 1 — Crear `src/api/compras.js`

Crear el archivo con todas las funciones API de compras. Seguir el mismo patrón de `src/api/http.js` (usar `request` y `json`).

```js
import { request, json } from './http.js'

// GET /compras → array de Compra[]
export async function getCompras() {
  const res = await request('/compras', { method: 'GET' })
  return json(res)
}

// GET /compras/{id} → Compra
export async function getCompra(id) {
  const res = await request(`/compras/${id}`, { method: 'GET' })
  return json(res)
}

// POST /compras → body: CreateCompraRequest → Compra (201)
export async function crearCompra(body) {
  const res = await request('/compras', { method: 'POST', body })
  return json(res)
}

// PUT /compras/{id} → body: CreateCompraRequest → Compra (200)
export async function actualizarCompra(id, body) {
  const res = await request(`/compras/${id}`, { method: 'PUT', body })
  return json(res)
}

// DELETE /compras/{id} → 204
export async function eliminarCompra(id) {
  const res = await request(`/compras/${id}`, { method: 'DELETE' })
  if (!res.ok) {
    const text = await res.text()
    throw new Error(text || 'Error al eliminar compra')
  }
  return null
}

// PUT /compras/{id}/imagen → multipart/form-data, parte "file"
export async function subirImagenCompra(id, file) {
  const formData = new FormData()
  formData.append('file', file)
  const res = await request(`/compras/${id}/imagen`, {
    method: 'PUT',
    body: formData,
  })
  if (!res.ok) {
    const text = await res.text()
    throw new Error(text || 'Error al subir imagen')
  }
  return null
}

// GET /compras/{id}/imagen → blob binario
export async function obtenerImagenCompra(id) {
  const res = await request(`/compras/${id}/imagen`, { method: 'GET' })
  if (!res.ok) {
    if (res.status === 404) return null
    throw new Error('Error al obtener imagen')
  }
  return res.blob()
}

// GET /compras/tipos-equipo → string[]
export async function getTiposEquipo() {
  const res = await request('/compras/tipos-equipo', { method: 'GET' })
  return json(res)
}
```

### Estructura general del `<script setup>`

Variables de estado:

```
// Datos
compras              → ref([])        — listado completo de compras
tiposEquipo          → ref([])        — enum TipoEquipo (cargado dinámicamente)

// UI states

error                → ref('')        — error en carga
toast                → ref('')        — mensaje de toast

// Filtros
searchText           → ref('')        — búsqueda texto libre
filtroTipo           → ref('TODOS')   — tipo de compra

filtroProveedor      → ref('TODOS')   — proveedor
filtroFechaDesde     → ref('')        — fecha desde
filtroFechaHasta     → ref('')        — fecha hasta

// Ordenamiento

// Modal crear/editar
formModal            → ref({ open: false, editing: null, ... })
    .open, .editing (null = crear, Compra = editar)
    .proveedorText, .proveedorId, .proveedorDropdownOpen
    .fechaCompra, .fechaFactura, .numeroFactura
    .importe, .moneda, .tipoCompra
    .tipoEquipo, .descripcionEquipo
    .descripcion, .observaciones
    .errors ({}), .apiError, .loading

// Modal detalle
detailModal          → ref({ open: false, compra: null, imageUrl: null, imageLoading: false, uploading: false })

// Modal confirmar eliminación
confirmModal         → ref({ open: false, compra: null, loading: false })

// Permisos
user                 → inject('dashboardUser')
isAdmin              → computed(() => user.value?.authorities?.includes('ROLE_ADMIN'))
```

Datos derivados (computed):

```
proveedoresUnicos    → deduplicar compras por proveedor.id → array { id, nombre, cuit }
monedasPresentes     → extraer monedas únicas → array de strings
filteredCompras      → aplicar TODOS los filtros (AND) sobre compras
sortedCompras        → aplicar ordenamiento sobre filteredCompras
totalPorMoneda       → agrupar sortedCompras por moneda → sumar importes → objeto { ARS: 125000, USD: 3200 }
cantidadFiltrada     → sortedCompras.length
tipoMasFrecuente     → contar por tipoCompra en sortedCompras → el que más tiene
hayFiltrosActivos    → true si algún filtro no está en su valor default
```

## PASO 5 — Validación del formulario

Función `validateForm()` que valida antes de `saveCompra()`:

```js
function validateForm() {
  const errors = {}
  const f = formModal.value
  
  if (!f.proveedorId && !f.proveedorText?.trim()) {
    errors.proveedor = 'Seleccioná o ingresá un proveedor.'
  }
  if (!f.fechaCompra) {
    errors.fechaCompra = 'La fecha de compra es obligatoria.'
  }
  if (!f.importe || parseFloat(f.importe) <= 0) {
    errors.importe = 'El importe debe ser mayor a 0.'
  }
  if (!f.tipoCompra) {
    errors.tipoCompra = 'Seleccioná un tipo de compra.'
  }
  if (f.tipoCompra === 'EQUIPO' && !f.tipoEquipo) {
    errors.tipoEquipo = 'Seleccioná un tipo de equipo.'
  }
  
  formModal.value.errors = errors
  return Object.keys(errors).length === 0
}
```

  }

## Notas importantes para el programador

1. **Referenciar UsuariosView como base de patrones.** Los estilos de tabla, modales, badges, toast, spinner y estados ya existen ahí. Copiar y adaptar.
2. **`inject('dashboardUser')`** para obtener el usuario y chequear `ROLE_ADMIN`. Viene de `DashboardLayout`.
3. **FormData para imagen:** `request()` en `http.js` ya detecta `FormData` y no serializa a JSON. Solo hacer `new FormData(); formData.append('file', file)`.
4. **Blob para imagen:** Para mostrar imagen del backend (que devuelve `application/octet-stream`), usar `res.blob()` (no `json()`) y `URL.createObjectURL(blob)`. Limpiar con `URL.revokeObjectURL()` al cerrar modal.
5. **Proveedor sin CRUD:** Extraer proveedores del listado de compras. Si no hay compras, el usuario solo puede crear proveedores nuevos.
6. **Site omitido:** No incluir `siteId` en el formulario (no hay endpoint para listar sites).
7. **TipoEquipo dinámico:** Cargar de `GET /compras/tipos-equipo`. No hardcodear.
8. **Moneda:** Usar `Intl.NumberFormat` para formatear. Default `ARS`.
9. **Fechas:** Enviar ISO `yyyy-MM-dd`, mostrar `dd/mm/yyyy`.
10. **Esta es la vista más importante del sistema.** Debe ser pulida, profesional y funcional. El código debe ser limpio y organizado.
